Learn Pseudocode

An interactive guide to the Pseudocode programming language. Work through these examples to learn the syntax and features.

The Basics

Pseudocode is designed to read like natural language. No semicolons, no braces — just clean, readable code.

hello.pseudo
// Your first Pseudocode program
print("Hello, World!")

Note: Comments use //. There are no multi-line comments.

Variables

Use let to declare variables. Types are inferred automatically.

variables.pseudo
let name = "Alice"          // String
let age = 25                 // Number
let pi = 3.14159             // Number (all numbers are 64-bit floats)
let active = true           // Boolean
let empty = nil             // Nil value

// Reassignment (no 'let' needed)
name = "Bob"
print(name)  // Bob

Data Types

Type Examples Description
Number 42, 3.14, -17 64-bit floating point
String "hello", 'world' Text (single or double quotes)
Boolean true, false Logical values
Nil nil Absence of value
Array [1, 2, 3] Ordered collection
Dictionary {"key": "value"} Key-value pairs

Operators

Arithmetic

let a = 10 + 5    // 15
let b = 10 - 5    // 5
let c = 10 * 5    // 50
let d = 10 / 5    // 2
let e = 10 % 3    // 1 (modulo)

Comparison

5 == 5     // true
5 != 3     // true
5 < 10     // true
5 > 10     // false
5 <= 5     // true
5 >= 6     // false

Logical

true and false   // false
true or false    // true
not true         // false

Conditionals

Use if/then/elif/else/end for conditional logic.

conditionals.pseudo
let score = 85

if score >= 90 then
    print("A")
elif score >= 80 then
    print("B")
elif score >= 70 then
    print("C")
else
    print("F")
end

Loops

For Loops

// Range-based (1 to 5)
for i in 1..6 do
    print(i)  // 1, 2, 3, 4, 5
end

// Iterate over array
let colors = ["red", "green", "blue"]
for color in colors do
    print(color)
end

While Loops

let n = 5
while n > 0 do
    print(n)
    n = n - 1
end

Pattern Matching

Use match/case for expressive conditional logic with optional guards.

match.pseudo
fn describe(n)
    match n
        case 0 then
            return "zero"
        case 1 then
            return "one"
        case x if x < 0 then
            return "negative"
        case _ then
            return "many"
    end
end

print(describe(0))   // zero
print(describe(-5))  // negative
print(describe(42))  // many

Tip: Use _ as a wildcard to match anything. Guards with if add extra conditions.

Defining Functions

Use fn to define functions.

functions.pseudo
fn greet(name)
    return "Hello, " + name + "!"
end

print(greet("World"))  // Hello, World!

// Multiple parameters
fn add(a, b)
    return a + b
end

print(add(3, 4))  // 7

Closures

Functions can capture variables from their enclosing scope.

closures.pseudo
fn make_counter()
    let count = 0
    return fn()
        count = count + 1
        return count
    end
end

let counter = make_counter()
print(counter())  // 1
print(counter())  // 2
print(counter())  // 3

Higher-Order Functions

Functions can be passed as arguments and returned from other functions.

fn apply_twice(f, x)
    return f(f(x))
end

fn double(n)
    return n * 2
end

print(apply_twice(double, 5))  // 20

Arrays

Ordered collections that can hold any type of value.

arrays.pseudo
let nums = [1, 2, 3, 4, 5]

// Access elements
print(nums[0])      // 1
print(nums[-1])     // 5 (last element)

// Modify
nums[0] = 10

// Built-in functions
push(nums, 6)       // Add to end
let last = pop(nums) // Remove from end
print(len(nums))    // Length

Dictionaries

Key-value pairs for storing structured data.

dictionaries.pseudo
let person = {
    "name": "Alice",
    "age": 30,
    "active": true
}

// Access values
print(person["name"])  // Alice

// Modify
person["age"] = 31

// Built-in functions
print(keys(person))    // ["name", "age", "active"]
print(values(person))  // ["Alice", 31, true]
print(has_key(person, "name"))  // true

Classes

Define your own types with methods and inheritance.

classes.pseudo
class Animal
    fn init(name)
        self.name = name
    end

    fn speak()
        return self.name + " makes a sound"
    end
end

class Dog extends Animal
    fn speak()
        return self.name + " says Woof!"
    end
end

let dog = Dog("Buddy")
print(dog.speak())  // Buddy says Woof!

File I/O

Read and write files with built-in functions.

// Read a file
let content = read_file("data.txt")
print(content)

// Write to a file
write_file("output.txt", "Hello, file!")

// Append to a file
append_file("log.txt", "New log entry\n")

// Check if file exists
if file_exists("config.json") then
    print("Found config!")
end

HTTP & JSON

Make HTTP requests and parse JSON with built-in functions.

http.pseudo
// GET request
let response = http_get("https://api.github.com/users/octocat")
let user = json_parse(response)
print(user["login"])  // octocat

// POST request
let data = json_stringify({"name": "test"})
let result = http_post("https://httpbin.org/post", data)
print(result)

Functional Programming

Use map, filter, and reduce for data transformation.

functional.pseudo
let nums = [1, 2, 3, 4, 5]

// Map: transform each element
let doubled = map(nums, fn(x) return x * 2 end)
print(doubled)  // [2, 4, 6, 8, 10]

// Filter: keep matching elements
let evens = filter(nums, fn(x) return x % 2 == 0 end)
print(evens)  // [2, 4]

// Reduce: combine into single value
let sum = reduce(nums, fn(acc, x) return acc + x end, 0)
print(sum)  // 15

// Chain operations
let result = reduce(
    filter(
        map(nums, fn(x) return x * 2 end),
        fn(x) return x > 4 end
    ),
    fn(acc, x) return acc + x end,
    0
)
print(result)  // 24 (6 + 8 + 10)
