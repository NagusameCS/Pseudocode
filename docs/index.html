See It In Action

Clean, readable code that does exactly what it looks like

fibonacci.pseudo 
// Recursive Fibonacci
fn fib(n)
    if n <= 1 then
        return n
    end
    return fib(n - 1) + fib(n - 2)
end

print(fib(30))  // 832040
fizzbuzz.pseudo 
for i in 1..100 do
    if i % 15 == 0 then
        print("FizzBuzz")
    elif i % 3 == 0 then
        print("Fizz")
    elif i % 5 == 0 then
        print("Buzz")
    else
        print(i)
    end
end
arrays.pseudo 
// Dynamic arrays
let nums = [1, 2, 3, 4, 5]

// Built-in functions
print(len(nums))     // 5
push(nums, 6)
print(pop(nums))     // 6

// Indexing
print(nums[0])       // 1
print(nums[-1])      // 5 (negative index)

Getting Started

1

Clone the Repository

Get the source code from GitHub

git clone https://github.com/NagusameCS/Pseudocode.git
2

Build the VM

Compile the high-performance C VM

cd Pseudocode/cvm && make
3

Run Your First Program

Execute a Pseudocode file

./pseudo ../examples/hello.pseudo
